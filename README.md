def calc(v1,v2,op):
  result = 0
  if op =='+':
    result = v1 + v2
  elif op == '-':
    result = v1 - v2
  elif op == '*':
    result = v1 * v2
  elif op == '/':
    result = v1 / v2
  return result
rst = 0
var1, var2, opr = 0,0,""
opr = input("(+,-,*,/) 입력: ")

var2 = int(input("2번째 수 입력: "))

#계산기 코드
def calc(v1,v2,op):
  result = 0
  if op =='+':
    result = v1 + v2
  elif op == '-':
    result = v1 - v2
  elif op == '*':
    result = v1 * v2
  elif op == '/':
    result = v1 / v2
  elif op == '**':
    result = v1 ** v2
  return result
rst = 0
var1, opr, var2 = 0,0,""
opr = input("(+,-,*,/,**) 입력: ")
var1 = int(input("1번째 수 입력: "))
var2 = int(input("2번째 수 입력: "))
rst = calc(var1,opr,var2)
